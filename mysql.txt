CREATE TABLE `cvddb`.`tblbanner`(
`Id` VARCHAR(100) NOT NULL , 
`Title` VARCHAR(100) NULL DEFAULT NULL , 
`Image` LONGTEXT NULL DEFAULT NULL , 
`Description` TEXT NULL DEFAULT NULL , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 

CREATE TABLE `cvddb`.`tblpredictionhistory`(
`Id` VARCHAR(100) NOT NULL , 
`UserId` VARCHAR(100) NULL DEFAULT NULL , 
`HeartDiseasesRiskProbabilty` DOUBLE NULL DEFAULT '0' , 
`StrokeRiskProbabilty` DOUBLE NULL DEFAULT '0' , 
`MentalDiseasesRiskProbabilty` DOUBLE NULL DEFAULT '0' , 
`CombinedRiskProbabilty` DOUBLE NULL DEFAULT '0' , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `cvddb`.`tbluser`(
`Id` VARCHAR(100) NOT NULL , 
`Name` VARCHAR(100) NULL DEFAULT NULL , 
`Username` VARCHAR(200) NULL DEFAULT NULL , 
`Password` VARCHAR(200) NULL DEFAULT NULL , 
`Phone` INT NULL DEFAULT NULL , 
`Active` VARCHAR(10) NULL DEFAULT NULL , 
`UserRole` VARCHAR(10) NULL DEFAULT NULL ,
`Weight` DECIMAL(19,4) NULL DEFAULT '0' , 
`Height` DECIMAL(19,4) NULL DEFAULT '0' , 
`BMI` DECIMAL(19,4) NULL DEFAULT '0' , 
`IsSmokingYes` BOOLEAN NULL DEFAULT FALSE , 
`IsAlcoholDrinkingYes` BOOLEAN NULL DEFAULT FALSE , 
`PhysicalHealth` INT NULL DEFAULT '0' ,
`MentalHealth` INT NULL DEFAULT '0' ,
`IsDiffWalkingYes` BOOLEAN NULL DEFAULT FALSE ,
`IsPhysicalActivityYes` BOOLEAN NULL DEFAULT FALSE ,
`GenHealth` INT NULL DEFAULT '0' ,
`SleepTime` INT NULL DEFAULT '0' ,
`IsHeartDiseaseYes` BOOLEAN NULL DEFAULT FALSE ,
`IsStrokeYes` BOOLEAN NULL DEFAULT FALSE ,
PRIMARY KEY (`Id`)) ENGINE = InnoDB;
