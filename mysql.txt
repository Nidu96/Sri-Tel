CREATE TABLE `agrodb`.`category`(
`Id` VARCHAR(100) NOT NULL , 
`Title` VARCHAR(100) NULL DEFAULT NULL , 
`Icon` LONGTEXT NULL DEFAULT NULL ,
`Description` TEXT NULL DEFAULT NULL , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `agrodb`.`banner`(
`Id` VARCHAR(100) NOT NULL , 
`Title` VARCHAR(100) NULL DEFAULT NULL , 
`Image` LONGTEXT NULL DEFAULT NULL , 
`Description` TEXT NULL DEFAULT NULL , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `agrodb`.`systemuser`(
`Id` VARCHAR(100) NOT NULL , 
`Name` VARCHAR(100) NULL DEFAULT NULL , 
`Username` VARCHAR(200) NULL DEFAULT NULL , 
`Password` VARCHAR(200) NULL DEFAULT NULL , 
`Active` VARCHAR(10) NULL DEFAULT NULL , 
`UserRole` VARCHAR(10) NULL DEFAULT NULL ,
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `agrodb`.`product`(
`Id` VARCHAR(100) NOT NULL , 
`CategoryId` VARCHAR(100) NULL DEFAULT NULL ,
`Title` VARCHAR(100) NULL DEFAULT NULL , 
`Image` LONGTEXT NULL DEFAULT NULL , 
`Price` DECIMAL(16,2) NULL DEFAULT NULL,
`Weight` INT NULL DEFAULT NULL , 
`Description` TEXT NULL DEFAULT NULL , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `agrodb`.`productorder`(
`Id` VARCHAR(100) NOT NULL , 
`IdForCustomer` VARCHAR(100) NOT NULL , 
`UserEmail` VARCHAR(100) NOT NULL , 
`DeliveryNote` VARCHAR(250) NOT NULL ,  
`Phone` VARCHAR(250) NOT NULL , 
`RecepientName` VARCHAR(250) NOT NULL , 
`RecepientPhone` VARCHAR(250) NOT NULL , 
`Status` VARCHAR(250) NOT NULL , 
`City` VARCHAR(100) NOT NULL ,
`DateOfPayment` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 

CREATE TABLE `agrodb`.`orderedproduct`(
`Id` VARCHAR(100) NULL DEFAULT NULL , 
`OrderId` VARCHAR(100) NULL DEFAULT NULL ,  
`Quantity` VARCHAR(100) NULL DEFAULT NULL , 
`ProductID` VARCHAR(250) NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB;
