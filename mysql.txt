CREATE TABLE `sriteldb`.`banner`(
`Id` VARCHAR(100) NOT NULL , 
`Title` VARCHAR(100) NULL DEFAULT NULL , 
`Image` LONGTEXT NULL DEFAULT NULL , 
`Description` TEXT NULL DEFAULT NULL , 
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 


CREATE TABLE `sriteldb`.`systemuser`(
`Id` VARCHAR(100) NOT NULL , 
`Name` VARCHAR(100) NULL DEFAULT NULL , 
`Username` VARCHAR(200) NULL DEFAULT NULL , 
`Password` VARCHAR(200) NULL DEFAULT NULL , 
`Phone` VARCHAR(200) NULL DEFAULT NULL , 
`Active` VARCHAR(10) NULL DEFAULT NULL , 
`UserRole` VARCHAR(10) NULL DEFAULT NULL ,
`Roaming` VARCHAR(10) NULL DEFAULT NULL , 
`RingingTone` VARCHAR(10) NULL DEFAULT NULL , 
`WorkPackage` VARCHAR(10) NULL DEFAULT NULL , 
`StudentPackage` VARCHAR(10) NULL DEFAULT NULL ,
`FamilyPackage` VARCHAR(10) NULL DEFAULT NULL ,
`FamilyPlusPackage` VARCHAR(10) NULL DEFAULT NULL ,
`WorkStudentPackage` VARCHAR(10) NULL DEFAULT NULL ,
PRIMARY KEY (`Id`)) ENGINE = InnoDB; 

CREATE TABLE `sriteldb`.`bills`(
`Id` VARCHAR(100) NOT NULL , 
`UserId` VARCHAR(100) NOT NULL , 
`TotalAmount` VARCHAR(100) NULL DEFAULT NULL , 
`Payment` VARCHAR(100) NULL DEFAULT NULL ,  
`DatePublished` DATETIME NULL DEFAULT NULL , 
PRIMARY KEY (`Id`)) ENGINE = InnoDB;
